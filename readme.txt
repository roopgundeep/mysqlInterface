There should be space between each and every string.
Examples
-> select * from countries ;
-> select CODE , ID from airports 
-> select distinct NAME from airports 
-> select * from airports where LATITUDE = LONGITUDE 
-> select LATITUDE from airports where LATITUDE >= LONGITUDE 
-> select * from airports where LATITUDE = 42 
-> select * from airports where NAME = heliport 
-> select * from countries , airports where ID = ID 
-> select * from airports where NAME = "HELLO"
